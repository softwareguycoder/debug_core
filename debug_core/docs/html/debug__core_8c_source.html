70  exit(ERROR);

72

74  // mutex handle for the function to initialize with NULL for the attributes.
75  int nResult = pthread_mutex_init(pMutex, NULL);

77  // Cleanup the mutex handle if necessary

80  pMutex = NULL;
81  }

85  }

252  g_fpLog = g_fpErrorLog = fopen(pszPath, "a+");
253  if (g_fpLog == NULL) {

257  }
258  }
